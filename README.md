Seccion 8: Contar las vocales con un for

cadena = "Hola Mundo"
vocales = 'aeiou'
cantidad = 0

for i in range(len(cadena)):
    if cadena[i].lower() in vocales:
        cantidad = cantidad + 1
    
print(f"La cantidad de vocales es {cantidad}")
